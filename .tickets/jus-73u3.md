# Implement cmd_proxy() command handler

Implement the proxy command handler that dispatches to service control functions.

## Function Implementation

```bash
cmd_proxy() {
    local action="${1:-status}"
    
    case "$action" in
        start)
            start_proxy_service
            ;;
        stop)
            stop_proxy_service
            ;;
        restart)
            restart_proxy_service
            ;;
        status)
            if is_proxy_running; then
                log_success "Proxy running"
                local count=$(jq 'length' "$JUSTVIBIN_PROJECTS_FILE" 2>/dev/null || echo 0)
                log_info "Serving $count project(s)"
            else
                log_warn "Proxy not running"
                log_info "Start with: justvibin proxy start"
            fi
            ;;
        logs)
            if [[ -f "$JUSTVIBIN_LOG" ]]; then
                tail -f "$JUSTVIBIN_LOG"
            else
                log_warn "No log file found"
            fi
            ;;
        *)
            log_error "Unknown action: $action"
            log_info "Usage: justvibin proxy [start|stop|restart|status|logs]"
            exit 1
            ;;
    esac
}
```

## Acceptance Criteria

- [ ] start/stop/restart/status/logs work
- [ ] status shows project count
- [ ] logs tails the log file
- [ ] Unknown action shows help
